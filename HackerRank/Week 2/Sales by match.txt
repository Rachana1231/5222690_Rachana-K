#include<stdio.h>
int sockMerchant(int n,int ar[]){
    int colors[101]={0};
    int pairs=0;
    for(int i=0;i<n;i++){
        colors[ar[i]]++;
        if(colors[ar[i]]==2){
            pairs++;
            colors[ar[i]]=0;
        }
    }
    return pairs;
}

    int n;
    scanf("%d",&n);
    int ar[n];
    for(int i=0;i<n;i++)
    scanf("%d",&ar[i]);
    printf("%d\n",sockMerchant(n,ar));
    return 0;
}
