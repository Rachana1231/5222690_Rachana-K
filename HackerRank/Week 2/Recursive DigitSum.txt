#include<stdio.h>
#include<string.h>
 long long digitSum(char *n){
    long long sum=0;
    for(int i=0;n[i]!='\0';i++){
        sum+=(n[i]-'0');
    }
    return sum;
 }
 int superDigitHelper(long long num){
    if(num<10) return num;
    long long sum=0;
    while (num>0){
        sum+=num%10;
        num/=10;
    }
    return superDigitHelper(sum);
 }
 int superDigit(char *n,int k){
    long long initialSum=digitSum(n);
    long long total=initialSum*k;
    return superDigitHelper(total);
 }
 int main(){
    char n[100005];
    int k;
    scanf("%s %d",n,&k);
    int result=superDigit(n,k);
    printf("%d\n",result);
    return 0;
 }
