#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int cmpfunc(const void*a,const void*b){
    return (*(char*)a - *(char*)b);
}
void gridChallenge(int n,char grid[n][n+1]){
    for(int i=0;i<n;i++){
        qsort(grid[i],n,sizeof(char),cmpfunc);
    }
    for(int col=0;col<n;col++){
        for(int row=1;row<n;row++){
           if(grid[row][col]<grid[row-1][col]){
            printf("NO\n");
            return;
           }
        }
    }
    printf("YES\n");
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        char grid[n][n+1];
        for(int i=0;i<n;i++){
            scanf("%s",grid[i]);
        }
        gridChallenge(n,grid);
    }
    return 0;
}
