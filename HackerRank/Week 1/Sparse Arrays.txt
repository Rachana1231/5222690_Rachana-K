Problem 4:(Sparse Arrays)
#include <stdio.h>
#include<string.h>

#define MAX_STRINGS 1000
#define MAX_LENGTH 21

int main(){
    int n,q;
    char strings[MAX_STRINGS][MAX_LENGTH];
    char queries[MAX_STRINGS][MAX_LENGTH];
    int count[MAX_STRINGS]={0};
    scanf("%d",&n);
    for (int i=0;i<n;i++){
        scanf("%s",strings[i]);
    }
    scanf("%d",&q);
    
    for(int i=0;i<q;i++){
        scanf("%s",queries[i]);
    }
    for(int i=0;i<q;i++){
        int occurrences=0;
        for(int j=0;j<n;j++){
            if(strcmp(queries[i],strings[j]) == 0){
                occurrences++;
            }
        }
        count[i]=occurrences;
    }
    for(int i=0;i<q;i++){
        printf("%d\n",count[i]);
    }
    return 0;
}
