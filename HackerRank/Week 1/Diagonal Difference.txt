Problem 7(Diagonal Difference)

#include<stdio.h>
#include<stdlib.h>
 int main(){
    int n;
    scanf("%d",&n);
    int arr[n][n];
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int primarySum=0,secondarySum=0;
    for (int i=0;i<n;i++){
        primarySum+=arr[i][i];
        secondarySum+=arr[i][n-1-i];
    }
    printf("%d\n",abs(primarySum-secondarySum));
    return 0;
 }
