Pangrams:

#include<stdio.h>
#include<ctype.h>
int main(){
    char s[1001],seen[26]={0};
    int c,count=0;
    fgets(s,sizeof(s),stdin);
    for(int i=0;s[i];i++){
        if(isalpha(s[i])){
            c=tolower(s[i])-'a';
            if(!seen[c]){
                seen[c]=1;
                count++;
            }
        }
    }
    printf(count==26?"pangram":"not pangram\n");
    return 0;
}
